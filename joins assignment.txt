DEPT_ID number primary key,

EMP_ID NUMBER,
EMP_NAME VARCHAR2(20),
SALARY NUMBER,
ANNUAL_BONUS NUMBER,
GENDER VARCHAR2(20),
DEPT_ID NUMBER,
CONSTRAINT PK_EMP PRIMARY KEY(EMP_ID),
CONSTRAINT FK_DEPTR_NUMB FOREIGN KEY (DEPT_Id) REFERENCES DEP (DEPT_ID));
CREATE TABLE LOC(
L_ID NUMBER,
L_NAME VARCHAR2(20));
